def cargar_plan_cuentas_db(df):
    conn = conectar_db()
    if conn:
        try:
            cursor = conn.cursor()
            
            # 1. Limpiamos los nombres de las columnas por si tienen espacios invisibles
            df.columns = [str(c).strip() for c in df.columns]
            
            # 2. Mapeamos los nombres exactos que veo en tu imagen
            # Si tu Excel dice 'tipo', lo dejamos así, pero si tiene mayúsculas lo corregimos
            df = df.rename(columns={
                'Nombre de la Cuenta': 'nombre',
                'Codigo': 'codigo'
            })

            cursor.execute("SET FOREIGN_KEY_CHECKS = 0;")
            cursor.execute("TRUNCATE TABLE plan_cuentas")
            cursor.execute("SET FOREIGN_KEY_CHECKS = 1;")
            
            for _, row in df.iterrows():
                # --- LÓGICA DE LIMPIEZA PARA TU EXCEL ESPECÍFICO ---
                
                # Extraer 'Grupo' o 'Detalle' del texto largo
                t_raw = str(row['tipo'])

                padre_raw = str(row['padre'])

                match_padre = re.search(r'(\d+(\.\d+)*)', padre_raw)
                padre_final = match_padre.group(1) if match_padre else None

                nombre_final = re.sub(r'^\d+(\.\d+)*\s*', '', nombre_raw).strip()

                query = """INSERT INTO plan_cuentas (id, codigo, nombre, nivel, tipo, padre) 
                           VALUES (%s, %s, %s, %s, %s, %s)"""
                cursor.execute(query, (
                    row['id'], codigo_final, nombre_final, 
                    row['nivel'], tipo_final, padre_final
                ))
            
            conn.commit()
            st.success("🚀 Plan de Cuentas cargado correctamente analizando tu formato.")

def limpiar_monto_contable(valor):
    # Convertimos a string y limpiamos espacios
    v = str(valor).strip()
    
    if v in ['-', '', 'nan', 'None', '0', '0.0']: 
        return 0.0
    
    try:
        # PASO A PASO PARA NO MULTIPLICAR POR 10:
        # Ejemplo: "147.659,00"
        v = v.replace('.', '')    # Queda "147659,00"
        v = v.replace(',', '.')    # Queda "147659.00"
        return float(v)
    except:
        return 0.0

def ejecutar_mayor_analitico(conn, cuenta, fecha_desde, fecha_hasta):
    # 1. SALDO BASE (Histórico)
    query_base = "SELECT SUM(debe - haber) as base FROM saldos_iniciales WHERE cuenta_contable = %s"
    df_base = pd.read_sql(query_base, conn, params=(cuenta,))
    saldo_base = df_base['base'].iloc[0] or 0

    # 2. MOVIMIENTOS PREVIOS (Antes de la fecha de inicio)
    query_previo = "SELECT SUM(debe - haber) as previo FROM asientos_contables WHERE cuenta_contable = %s AND fecha < %s"
    df_prev = pd.read_sql(query_previo, conn, params=(cuenta, fecha_desde))
    saldo_previo = df_prev['previo'].iloc[0] or 0
    
    saldo_inicial_periodo = saldo_base + saldo_previo

    # 3. MOVIMIENTOS DEL PERIODO (Lo que quieres sumar)
    query_movs = "SELECT fecha, n_comprobante, descripcion, referencia, debe, haber FROM asientos_contables WHERE cuenta_contable = %s AND fecha BETWEEN %s AND %s ORDER BY fecha ASC, id ASC"
    df_movs = pd.read_sql(query_movs, conn, params=(cuenta, fecha_desde, fecha_hasta))

    # 4. FILA DE SALDO INICIAL (Para que aparezca de primero en la tabla)
    fila_inicial = pd.DataFrame([{
        'fecha': fecha_desde, 
        'n_comprobante': 'S/I', 
        'descripcion': f'SALDO INICIAL AL {fecha_desde}', 
        'referencia': 'INICIAL', 
        'debe': 0.00, 
        'haber': 0.00, 
        'Saldo': saldo_inicial_periodo
    }])

    # 5. CÁLCULO LÍNEA POR LÍNEA
    saldos_lista = []
    acumulado = saldo_inicial_periodo
    for _, row in df_movs.iterrows():
        acumulado += (row['debe'] - row['haber'])
        saldos_lista.append(acumulado)
    df_movs['Saldo'] = saldos_lista

    # Unimos la fila del saldo inicial con los movimientos
    df_final = pd.concat([fila_inicial, df_movs], ignore_index=True)
    
    # IMPORTANTE: Devolvemos df_final (para la tabla) y df_movs (para los totales)
    return df_final, df_movs


def generar_balance_comprobacion(conn, f_i, f_f, sucursal):
    # 1. Protección contra selección vacía
    if not sucursal:
        return pd.DataFrame(columns=['Código', 'Debe', 'Haber', 'Saldo Inicial', 'Saldo Final'])
    
    # 2. Query de Saldos Iniciales (Global, sin el centro_costo que da error)
    query_si = """
        SELECT plan_cuentas, cuenta_contable, SUM(debe) - SUM(haber) as saldo_fijo 
        FROM saldos_iniciales 
        GROUP BY plan_cuentas, cuenta_contable
    """
    df_si = pd.read_sql(query_si, conn)

    # 3. Query de Acumulado Previo (Todo lo que pasó antes del mes actual)
    query_acum = """
        SELECT plan_cuentas, SUM(debe) - SUM(haber) as acumulado_previo 
        FROM asientos_contables 
        WHERE fecha < %s 
        GROUP BY plan_cuentas
    """
    df_acum = pd.read_sql(query_acum, conn, params=(f_i,))

    # 4. Query del Mes (Lo que pasó ESPECÍFICAMENTE en el mes seleccionado)
    query_mes = """
        SELECT plan_cuentas, cuenta_contable, 
               SUM(debe) as debe_mes, 
               SUM(haber) as haber_mes 
        FROM asientos_contables 
        WHERE fecha BETWEEN %s AND %s 
        GROUP BY plan_cuentas, cuenta_contable
    """
    df_movs = pd.read_sql(query_mes, conn, params=(f_i, f_f))

    # --- UNIÓN Y LÓGICA DE NEGOCIO ---
    # Unimos los movimientos del mes con los saldos iniciales fijos
    balance = pd.merge(df_movs, df_si, on=['plan_cuentas', 'cuenta_contable'], how='outer').fillna(0)
    # Unimos con el acumulado previo (por código de cuenta)
    balance = pd.merge(balance, df_acum, on='plan_cuentas', how='outer').fillna(0)
    
    balance['Tipo'] = balance['plan_cuentas'].astype(str).str[0]
    
    def ajustar_valores(row):
        # Saldo Inicial Real = Saldo de Apertura + Todo el arrastre del pasado
        si_bruto = row['saldo_fijo'] + row['acumulado_previo']
        
        if row['Tipo'] == '1': # ACTIVO
            saldo_inicial = si_bruto
            saldo_final = saldo_inicial + row['debe_mes'] - row['haber_mes']
        else: # PASIVO / PATRIMONIO / INGRESOS (Naturaleza Acreedora)
            saldo_inicial = abs(si_bruto) 
            saldo_final = saldo_inicial + row['haber_mes'] - row['debe_mes']
            
        return pd.Series([saldo_inicial, saldo_final])

    # Aplicamos el cálculo a las columnas finales
    balance[['Saldo Inicial', 'Saldo Final']] = balance.apply(ajustar_valores, axis=1)
    
    return balance.rename(columns={'plan_cuentas': 'Código', 'debe_mes': 'Debe', 'haber_mes': 'Haber'})

def estilo_balance(row):
    """Aplica colores y negritas según el nivel de la cuenta"""
    # Nivel 1: Títulos principales (Activo, Pasivo, etc.) - Azul Oscuro
    if row['nivel'] == 1:
        return ['background-color: #1a5276; color: white; font-weight: bold'] * len(row)
    
    # Nivel 2: Sub-títulos (Activo Corriente, etc.) - Azul Claro
    elif row['nivel'] == 2:
        return ['background-color: #d4e6f1; color: black; font-weight: bold'] * len(row)
    
    # Nivel 3 y 4: Grupos intermedios - Solo Negrita
    elif row['nivel'] in [3, 4]:
        return ['font-weight: bold'] * len(row)
    
    # Nivel 5: Cuentas de detalle (Caja, Bancos) - Normal
    return [''] * len(row)

                        if pd.isna(v) or str(v).strip() in ['nan', 'None', '']: return 0.0
                        res = float(str(v).replace('.', '').replace(',', '.'))
                        # Pararrayos para el error 'Out of range' (1600 -> 16.0)
                        return res / 100 if es_iva and res > 100 else res
                    except: return 0.0

                # Formateo de Factura (Columna C -> índice 2) y Control (Col D -> índice 3)
                n_fact = str(fila[2]).split('.')[0].strip().zfill(5)
                n_cont = str(fila[3]).split('.')[0].strip().zfill(5)

                # SQL REPLACE: Si la factura ya existe, la actualiza; si no, la crea.
                # Esto soluciona el error de "Duplicate Entry"
                sql = """REPLACE INTO libro_compras 
                         (fecha_operacion, tipo_documento, n_factura, n_control, proveedor, rif, 
                          total_compras, importe_exento, base_imponible, iva_porcentaje, iva_monto) 
                         VALUES (%s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s)"""

                # Procesar Fecha con respaldo si falla el formato
                try:
                    f_op = pd.to_datetime(fila[0]).strftime('%Y-%m-%d')
                except:
                    f_op = "2026-08-15" # Fecha por defecto basada en tu ejemplo

                valores = (
                    f_op,
                    str(fila[1]).split('.')[0].zfill(2), # Tipo Doc (Col B)
                    n_fact,
                    n_cont,
                    str(fila[4]).upper()[:255].strip(),    # Proveedor (Col E)
                    rif_raw.replace('-', '').replace('.', ''), # RIF limpio
                    f_n(fila[6]), f_n(fila[7]), f_n(fila[8]), # Montos (Col G, H, I)
                    f_n(fila[9], True), # Alicuota % (Col J)
                    f_n(fila[10])        # IVA (Col K)
                )

                cursor.execute(sql, valores)
                exitos += 1
                print(f"✅ Fila {i}: Factura {n_fact} cargada exitosamente.")

                print(f"❌ Error procesando fila {i}: {e}")

        cursor.close()

        if exitos > 0:

def borrar_compras_por_rango(desde, hasta):
    try:
        conexion = conectar_db()
        cursor = conexion.cursor()
        
        # 1. PRIMERO: Borramos las retenciones de ISLR que dependen de esas compras
        # Usamos un subquery para encontrar los IDs de las compras en ese rango
        sql_islr = """
            DELETE FROM retenciones_islr 
            WHERE id_compra IN (
                SELECT id FROM libro_compras 
                WHERE fecha_operacion BETWEEN %s AND %s
            )
        """
        cursor.execute(sql_islr, (desde, hasta))
        filas_islr = cursor.rowcount

        # 2. SEGUNDO: Ahora que no hay dependencias, borramos las compras
        sql_compras = "DELETE FROM libro_compras WHERE fecha_operacion BETWEEN %s AND %s"
        cursor.execute(sql_compras, (desde, hasta))
        filas_compras = cursor.rowcount
        
        conexion.commit()
        
        cursor.close()
        conexion.close()
        
        if filas_compras > 0:
            st.success(f"✅ Limpieza profunda completada:")
            st.write(f"* Compras eliminadas: {filas_compras}")
            st.write(f"* Retenciones ISLR eliminadas: {filas_islr}")
        else:
            st.warning("No se encontraron registros para borrar en ese rango.")
            
    except Exception as e:
        st.error(f"Error de Integridad: {e}")

def borrar_ventas_por_rango(desde, hasta):

def preparar_excel_descarga(df):
    output = io.BytesIO()
    with pd.ExcelWriter(output, engine='xlsxwriter') as writer:
        df.to_excel(writer, index=False, sheet_name='LibroDeVentas')
        # Ajuste automático de columnas
        workbook = writer.book
        worksheet = writer.sheets['LibroDeVentas']
        for i, col in enumerate(df.columns):
            column_len = max(df[col].astype(str).map(len).max(), len(col)) + 2
            worksheet.set_column(i, i, column_len)
    return output.getvalue()

def generar_comprobante_pdf(datos):
    """
    Crea el PDF del comprobante ISLR con diseño profesional simétrico,
    limpieza de etiquetas, RIF con guiones, número de comprobante legal 
    y centrado de celdas.
    """
    pdf = FPDF(orientation='P', unit='mm', format='A4')
    pdf.add_page()
    
    # --- 0. PREPARACIÓN DE DATOS (LIMPIEZA) ---
    import re
    from datetime import datetime

    def limpiar_num(texto):
        nums = re.findall(r'\d+', str(texto))
        return nums[0] if nums else str(texto)

    # Lógica para formatear RIF con guiones (V-12345678-9)
    def formatear_rif(rif_raw):
        rif = str(rif_raw).upper().replace('-', '').replace(' ', '')
        if len(rif) >= 9:
            return f"{rif[0]}-{rif[1:-1]}-{rif[-1]}"
        return rif

    factura_limpia = limpiar_num(datos.get('factura', '00000')).zfill(5)
    
    # --- 1. ENCABEZADO CORPORATIVO ---
    pdf.set_font("helvetica", "B", 10)
    # Nombre de la empresa a la izquierda
    pdf.cell(100, 5, datos['agente']['nombre'].upper(), 0, 0, 'L')
    
    # NÚMERO DE COMPROBANTE LEGAL (Derecha, resaltado)
    pdf.set_font("helvetica", "B", 11)

def obtener_siguiente_comprobante(self):
    from datetime import datetime
    
    # 1. Sacamos el prefijo del periodo actual (Ej: 202604)
    periodo_actual = datetime.now().strftime('%Y%m')
    
    # 2. Buscamos el último comprobante guardado en la DB que empiece con este año/mes
    # SQL: SELECT n_comprob_islr FROM retenciones_islr WHERE n_comprob_islr LIKE '202604%' ORDER BY n_comprob_islr DESC LIMIT 1
    ultimo_registro = self.db.query(
        "SELECT n_comprob_islr FROM retenciones_islr WHERE n_comprob_islr LIKE %s ORDER BY n_comprob_islr DESC LIMIT 1",
        (f"{periodo_actual}%",)
    ).fetchone()

    if ultimo_registro:
        # Si existe, tomamos los últimos 8 dígitos, le sumamos 1 y rellenamos con ceros
        ultimo_num = int(ultimo_registro[0][6:]) # Cortamos después de '202604'
        nuevo_correlativo = str(ultimo_num + 1).zfill(8)
    else:
        # Si es la primera retención del mes, empezamos en 1
        nuevo_correlativo = "00000001"

    return f"{periodo_actual}{nuevo_correlativo}"


def generar_xml_seniat(df, rif_agente, periodo):
    """
    Crea el string XML basado en los registros de retenciones filtrados.
    """
    # Nodo principal
    root = ET.Element("RelacionRetencionesISLR")
    root.set("RifAgente", rif_agente.replace("-", ""))
    root.set("Periodo", periodo)

    for _, row in df.iterrows():
        detalle = ET.SubElement(root, "DetalleRetencion")
        
        # RIF del retenido (sin guiones)
        ET.SubElement(detalle, "RifRetenido").text = str(row['rif_proveedor']).replace("-", "")
        # Número de Factura
        ET.SubElement(detalle, "NumeroFactura").text = str(row['n_factura'])
        # Número de Control
        ET.SubElement(detalle, "NumeroControl").text = str(row['n_control'])
        # Código de Concepto (ej: 001)
        ET.SubElement(detalle, "CodigoConcepto").text = str(row['codigo_conceptos']).zfill(3)
        # Base Imponible
        ET.SubElement(detalle, "MontoOperacion").text = f"{row['base_imponible']:.2f}"
        # Porcentaje de Retención
        ET.SubElement(detalle, "PorcentajeRetencion").text = f"{row['porcentaje_retencion']:.2f}"
        
    # Formatear el XML para que sea legible (pretty print)
    xml_str = ET.tostring(root, encoding='utf-8')
    parsed = minidom.parseString(xml_str)
    return parsed.toprettyxml(indent="  ")

    """

    """
    import pymysql

                razon_social = VALUES(razon_social),
                direccion_fiscal = VALUES(direccion_fiscal)
            """

            cursor.execute(sql, (rif, tipo, row['razon_social'], row['direccion_fiscal']))
        
        conn.commit()

def actualizar_tabla_completa_db(nombre_tabla, df_nuevo):
    conn = conectar_db()
    if conn:
        try:
            cursor = conn.cursor()
            cursor.execute("SET FOREIGN_KEY_CHECKS = 0;")
            cursor.execute("SET SQL_SAFE_UPDATES = 0;")
            cursor.execute(f"TRUNCATE TABLE {nombre_tabla}")
            
            for _, row in df_nuevo.iterrows():
                if pd.notna(row['rif']) and str(row['rif']).strip() != "":
                    # AGREGAMOS tipo_persona AL SQL
                    sql = f"INSERT INTO {nombre_tabla} (rif, tipo_persona, razon_social, direccion_fiscal) VALUES (%s, %s, %s, %s)"
                    cursor.execute(sql, (
                        str(row['rif']).upper(), 
                        row.get('tipo_persona', 'PJ'), # Por defecto PJ si viene vacío
                        row['razon_social'], 
                        row['direccion_fiscal']
                    ))
            
            conn.commit()
            cursor.execute("SET SQL_SAFE_UPDATES = 1;")
            cursor.execute("SET FOREIGN_KEY_CHECKS = 1;")
            conn.close()
        except Exception as e:
            if conn: conn.close()
            raise e

def generar_balance_profesional(conn, f_i, f_f, sucursal):
    """
    Toma los saldos del balance de comprobación y los suma hacia arriba
    según la jerarquía del Plan de Cuentas (Nivel 4 -> 3 -> 2 -> 1).
    """
    # 1. Obtener los saldos base (detalle por cuenta)
    df_saldos = generar_balance_comprobacion(conn, f_i, f_f, sucursal)
    
    # 2. Traer la estructura completa del Plan de Cuentas de la DB
    df_plan = pd.read_sql("SELECT codigo, nombre, nivel, padre FROM plan_cuentas ORDER BY codigo", conn)
    
    # 3. Preparar columnas numéricas para que no den error al sumar
    cols_numericas = ['Saldo Inicial', 'Debe', 'Haber', 'Saldo Final']
    for col in cols_numericas:
        if col in df_saldos.columns:
            df_saldos[col] = pd.to_numeric(df_saldos[col], errors='coerce').fillna(0)
        else:
            df_saldos[col] = 0.0

    # 4. Unir estructura (Plan) con movimientos (Saldos)
    df = pd.merge(df_plan, df_saldos, left_on='codigo', right_on='Código', how='left')
    df[cols_numericas] = df[cols_numericas].fillna(0)
    
    # 5. LÓGICA DE ROLL-UP (Suma jerárquica: el hijo le suma al padre)
    # Ordenamos los niveles de mayor a menor (ej. del 4 al 2)
    niveles = sorted(df['nivel'].unique(), reverse=True)
    
    for n in niveles:
        if n == 1: continue  # El nivel 1 es el tope, no tiene a quién sumarle
        
        # Filtramos las cuentas del nivel actual que tienen un padre
        df_hijos = df[(df['nivel'] == n) & (df['padre'].fillna('') != '')]
        
        for _, row in df_hijos.iterrows():
            # Buscamos al padre de esta cuenta
            padre_mask = df['codigo'] == row['padre']
            if padre_mask.any():
                # Le transferimos el saldo del hijo al padre
                df.loc[padre_mask, cols_numericas] += row[cols_numericas].values

    # 6. Limpiar columnas duplicadas del merge
    if 'Código' in df.columns:
        df = df.drop(columns=['Código'])
        
    return df

    return {

def obtener_datos_graficos(conn, f_i, f_f, sucursal):
    # 1. Inicializamos para evitar el NameError
    df_bar = pd.DataFrame(columns=['Categoría', 'Monto'])
    df_pie = pd.DataFrame(columns=['nombre', 'Saldo Final'])
    
    try:
        # --- QUERY PARA BARRAS (Sin el filtro de sucursal que da error) ---
        query_bar = """
            SELECT 
                CASE 
                    WHEN SUBSTRING(plan_cuentas, 1, 1) = '4' THEN 'Ingresos' 
                    WHEN SUBSTRING(plan_cuentas, 1, 1) = '5' THEN 'Egresos' 
                END as Categoría,
                ABS(SUM(debe - haber)) as Monto
            FROM asientos_contables
            WHERE fecha BETWEEN %s AND %s 
            AND SUBSTRING(plan_cuentas, 1, 1) IN ('4', '5')
            GROUP BY Categoría
        """
        # Ejecutamos con solo las dos fechas como parámetros
        df_bar = pd.read_sql(query_bar, conn, params=(f_i, f_f))

        # --- QUERY PARA LA DONA (Top 5 Gastos) ---
        query_pie = """
            SELECT plan_cuentas as nombre, ABS(SUM(debe - haber)) as `Saldo Final`
            FROM asientos_contables
            WHERE fecha BETWEEN %s AND %s 
            AND SUBSTRING(plan_cuentas, 1, 1) = '5'
            GROUP BY plan_cuentas
            ORDER BY `Saldo Final` DESC
            LIMIT 5
        """
        df_pie = pd.read_sql(query_pie, conn, params=(f_i, f_f))
        
    except Exception as e:
        # Esto te dirá exactamente qué pasa si algo falla
        st.error(f"Error técnico en gráficos: {e}")

    return df_bar, df_pie




def mostrar_treemap_gastos(conn, f_i, f_f):
    # Buscamos solo cuentas de gasto (nivel 5) con movimiento
    query = """
        SELECT plan_cuentas as Cuenta, 
               SUM(debe) - SUM(haber) as Monto 
        FROM asientos_contables 
        WHERE fecha BETWEEN %s AND %s AND SUBSTRING(plan_cuentas, 1, 1) = '5'
        GROUP BY plan_cuentas
        HAVING Monto > 0
    """
    df_gastos = pd.read_sql(query, conn, params=(f_i, f_f))
    
    if not df_gastos.empty:

                                # Nota: He mantenido los nombres exactos de tu DB (ej. 'Total_Comrpas' y 'Numero_Contro')
                                m_total = f"{float(reg['Total_Comrpas'] or 0):.2f}"
                                m_base = f"{float(reg['Base_Imponible'] or 0):.2f}"
                                m_ret = f"{float(reg['IVA_Retenido'] or 0):.2f}"
                                m_exento = f"{float(reg['Compras_Excentas'] or 0):.2f}"
                                m_ali = f"{float(16.00):.2f}"
                                # 4. CONSTRUCCIÓN DEL TXT PARA EL SENIAT
                                campos = [
                                    rif_agente,                                     # 1. RIF Agente
                                    periodo,                                        # 2. Periodo (AAAAMM)
                                    f_obj.strftime('%Y-%m-%d'),                     # 3. Fecha Factura (YYYY-MM-DD)
                                    'C',                                            # 4. Operación (Compras)

                                    '0'                                             # 16. Expediente
                                ]
                                
                                # Unimos con TABULACIÓN (\t) según exige el SENIAT
                                lineas_txt.append("\t".join(campos))


                            # --- ESTO YA ESTÁ FUERA DEL BUCLE 'FOR' ---
                            # Juntamos todas las líneas procesadas
                            contenido_final_txt = "\n".join(lineas_txt)
                            
                            st.markdown("### 📋 Vista Previa (Formato de Longitud Fija):")
                            st.code(contenido_final_txt)

                            # Definimos el nombre con las fechas elegidas

                            # Generamos un ID de tiempo para que la 'key' sea siempre única

                                use_container_width=False,

def generar_excel_formateado(df, titulo, subtitulo):
    output = io.BytesIO()
    
    # Usamos xlsxwriter como motor para manejar estilos fácilmente
    with pd.ExcelWriter(output, engine='xlsxwriter') as writer:
        df.to_excel(writer, index=False, sheet_name='Balance', startrow=4)
        
        workbook  = writer.book
        worksheet = writer.sheets['Balance']

        # 1. Definir Formatos (Colores similares a tu interfaz)
        formato_titulo = workbook.add_format({
            'bold': True, 'size': 16, 'font_color': '#1f2937', 'align': 'left'
        })
        formato_subtitulo = workbook.add_format({
            'bold': True, 'size': 12, 'font_color': '#4b5563', 'align': 'left'
        })
        formato_encabezado = workbook.add_format({
            'bold': True, 'text_wrap': True, 'valign': 'vcenter',
            'fg_color': '#1e3a8a', 'font_color': 'white', 'border': 1
        })

        # 2. Escribir Título y Subtítulo
        worksheet.write('A1', titulo, formato_titulo)
        worksheet.write('A2', subtitulo, formato_subtitulo)

        # 3. Aplicar color a los encabezados de la tabla (Fila 5 en Excel, índice 4)
        for col_num, value in enumerate(df.columns.values):
            worksheet.write(4, col_num, value, formato_encabezado)
            # Ajustar ancho de columna automáticamente
            worksheet.set_column(col_num, col_num, 20)

    return output.getvalue()
